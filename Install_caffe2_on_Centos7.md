> cmake3 -DBUILD_SHARED_LIBS=ON -DCMAKE_CXX_FLAGS='-fPIC' .. && \
> make -j 8 && sudo make install && cd ../.. && \
> git clone https://github.com/google/glog && \
> cd glog && \
> mkdir build && cd build && \
> cmake3 -DBUILD_SHARED_LIBS=ON -DCMAKE_CXX_FLAGS='-fPIC' .. && \
